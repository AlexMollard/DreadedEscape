          github-token: ${{ secrets.GITHUB_TOKEN }}

      - name: Build Project
        run: |
          # Enter commands to build your Unreal Engine project here

          ./Engine/Build/BatchFiles/RunUAT.bat BuildCookRun -project="./DreadedEscape.uproject" -platform=Win64 -clientconfig=Development -serverconfig=Development -cook -allmaps -build -stage -pak -archive -archivedirectory="./Build/"
